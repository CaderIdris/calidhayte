  40def cont_strat_folds(
  41        df: pd.DataFrame,
  42        target_var: str,
  43        splits: int = 5,
  44        strat_groups: int = 5,
  45        seed: int = 62
  46        ) -> pd.DataFrame:
  47    """
  48    Creates stratified k-folds on continuous variable.
  49    Based off of work by tolgadincer [^kaggle].
  50
  51    [^kaggle]:
  52    www.kaggle.com/code/tolgadincer/continuous-target-stratification/notebook
  53
  54    Parameters
  55    ----------
  56    df : pd.DataFrame
  57        Target data to stratify on.
  58    target_var : str
  59        Target feature name.
  60    splits : int, default=5
  61        Number of folds to make.
  62    strat_groups : int, default=10
  63        Number of groups to split data in to for stratification.
  64    seed : int, default=62
  65        Random state to use.
  66    
  67    Returns
  68    -------
  69    pd.DataFrame
  70        `y_df` with added 'Fold' column, specifying which test data fold
  71        variable corresponds to.
  72
  73    Examples
  74    --------
  75    >>> df = pd.read_csv('data.csv')
  76    >>> df
  77    |    | x | a | b |
  78    |    |   |   |   |
  79    |  0 |2.3|1.8|7.2|
  80    |  1 |3.2|9.6|4.5|
  81    |....|...|...|...|
  82    |1000|2.3|4.5|2.2|
  83    >>> df_with_folds = const_strat_folds(
  84            df=df,
  85            target='a',
  86            splits=3,
  87            strat_groups=3.
  88            seed=78
  89        )
  90    >>> df_with_folds
  91    |    | x | a | b |Fold|
  92    |    |   |   |   |    |
  93    |  0 |2.3|1.8|7.2| 2  |
  94    |  1 |3.2|9.6|4.5| 1  |
  95    |....|...|...|...|....|
  96    |1000|2.3|4.5|2.2| 0  |
  97
  98    All folds should have a roughly equal distribution of values for 'a'
  99
 100    """
 101    _df = df.copy()
 102    _df['Fold'] = -1
 103    skf = StratifiedKFold(
 104            n_splits=splits,
 105            random_state=seed,
 106            shuffle=True
 107            )
 108    _df['Group'] = pd.cut(
 109            _df.loc[:, target_var],
 110            strat_groups,
 111            labels=False
 112            )
 113    group_label = _df.loc[:, 'Group']
 114
 115    for fold_number, (_, v) in enumerate(skf.split(group_label, group_label)):
 116        _df.loc[v, 'Fold'] = fold_number
 117    return _df.drop('Group', axis=1)

 121    """
 122    Calibrate x against y using a range of different methods provided by
 123    scikit-learn[^skl], xgboost[^xgb] and PyMC (via Bambi)[^pmc].
 124
 125    [^skl]: https://scikit-learn.org/stable/modules/classes.html
 126    [^xgb]: https://xgboost.readthedocs.io/en/stable/python/python_api.html
 127    [^pmc]: https://bambinos.github.io/bambi/api/

 130    def __init__(
 131            self,
 132            x_data: pd.DataFrame,
 133            y_data: pd.DataFrame,
 134            target: str,
 135            folds: int = 5,
 136            strat_groups: int = 10,
 137            scalers: Union[
 138                Iterable[
 139                    Literal[
 140                        'None',
 141                        'Standard Scale',
 142                        'MinMax Scale',
 143                        'Yeo-Johnson Transform'
 144                        'Box-Cox Transform',
 145                        'Quantile Transform (Uniform)',
 146                        'Quantile Transform (Gaussian)'
 147                        ]
 148                    ],
 149                    Literal[
 150                        'All',
 151                        'None',
 152                        'Standard Scale',
 153                        'MinMax Scale',
 154                        'Yeo-Johnson Transform'
 155                        'Box-Cox Transform',
 156                        'Quantile Transform (Uniform)',
 157                        'Quantile Transform (Gaussian)',
 158                        ]
 159                ] = 'None',
 160            seed: int = 62
 161                 ):
 162        """Initialises class
 163
 164        Used to compare one set of measurements against another.
 165        It can perform both univariate and multivariate regression, though
 166        some techniques can only do one or the other. Multivariate regression
 167        can only be performed when secondary variables are provided.
 168
 169        Parameters
 170        ----------
 171        x_data : pd.DataFrame
 172            Data to be calibrated.
 173        y_data : pd.DataFrame
 174            'True' data to calibrate against.
 175        target : str
 176            Column name of the primary feature to use in calibration, must be
 177            the name of a column in both `x_data` and `y_data`.
 178        folds : int, default=5
 179            Number of folds to split the data into, using stratified k-fold.
 180        strat_groups : int, default=10
 181            Number of groups to stratify against, the data will be split into
 182            n equally sized bins where n is the value of `strat_groups`.
 183        scaler : iterable of {<br>\
 184            'None',<br>\
 185            'Standard Scale',<br>\
 186            'MinMax Scale',<br>\
 187            'Yeo-Johnson Transform',<br>\
 188            'Box-Cox Transform',<br>\
 189            'Quantile Transform (Uniform)',<br>\
 190            'Quantile Transform (Gaussian)',<br>\
 191            } or {<br>\
 192            'All',<br>\
 193            'None',<br>\
 194            'Standard Scale',<br>\
 195            'MinMax Scale',<br>\
 196            'Yeo-Johnson Transform',<br>\
 197            'Box-Cox Transform',<br>\
 198            'Quantile Transform (Uniform)',<br>\
 199            'Quantile Transform (Gaussian)',<br>\
 200            }, default='None'
 201            The scaling/transform method (or list of methods) to apply to the
 202            data
 203        seed : int, default=62
 204            Random state to use when shuffling and splitting the data into n
 205            folds. Ensures repeatability.
 206
 207        Raises
 208        ------
 209        ValueError
 210            Raised if the target variables (e.g. 'NO2') is not a column name in
 211            both dataframes.
 212
 213        Examples
 214        --------
 215        >>> from calidhayte.calibrate import Calibrate
 216        >>> import pandas as pd
 217        >>>
 218        >>> x = pd.read_csv('independent.csv')
 219        >>> x
 220        |   | a | b |
 221        | 0 |2.3|3.2|
 222        | 1 |3.4|3.1|
 223        |...|...|...|
 224        |100|3.7|2.1|
 225        >>>
 226        >>> y = pd.read_csv('dependent.csv')
 227        >>> y
 228        |   | a |
 229        | 0 |7.8|
 230        | 1 |9.9|
 231        |...|...|
 232        |100|9.5|
 233        >>>
 234        >>> calibration = Calibrate(
 235            x_data=x,
 236            y_data=y,
 237            target='a',
 238            folds=5,
 239            strat_groups=5,
 240            scaler = [
 241                'Standard Scale',
 242                'MinMax Scale'
 243                ],
 244            seed=62
 245        )
 246        >>> calibration.linreg()
 247        >>> calibration.lars()
 248        >>> calibration.omp()
 249        >>> calibration.ransac()
 250        >>> calibration.random_forest()
 251        >>>
 252        >>> models = calibration.return_models()
 253        >>> list(models.keys())
 254        [
 255            'Linear Regression',
 256            'Least Angle Regression',
 257            'Orthogonal Matching Pursuit',
 258            'RANSAC',
 259            'Random Forest'
 260        ]
 261        >>> list(models['Linear Regression'].keys())
 262        ['Standard Scale', 'MinMax Scale']
 263        >>> list(models['Linear Regression']['Standard Scale'].keys())
 264        ['a', 'a + b']
 265        >>> list(models['Linear Regression']['Standard Scale']['a'].keys())
 266        [0, 1, 2, 3, 4]
 267        >>> type(models['Linear Regression']['Standard Scale']['a'][0])
 268        <class sklearn.pipeline.Pipeline>
 269        >>> pipeline = models['Linear Regression']['Standard Scale']['a'][0]
 270        >>> x_new = pd.read_csv('independent_new.csv')
 271        >>> x_new
 272        |   | a | b |
 273        | 0 |3.5|2.7|
 274        | 1 |4.0|1.1|
 275        |...|...|...|
 276        |100|2.3|2.1|
 277        >>> pipeline.transform(x_new)
 278        |   | a |
 279        | 0 |9.7|
 280        | 1 |9.1|
 281        |...|...|
 282        |100|6.7|
 283
 284
 285        """
 286        if target not in x_data.columns or target not in y_data.columns:
 287            raise ValueError(
 288                    f"{target} does not exist in both columns."
 289                             )
 290        join_index = x_data.join(
 291                y_data,
 292                how='inner',
 293                lsuffix='x',
 294                rsuffix='y'
 295                ).dropna().index
 296        """
 297        The common indices between `x_data` and `y_data`, excluding missing
 298        values
 299        """
 300        self.x_data: pd.DataFrame = x_data.loc[join_index, :]
 301        """
 302        The data to be calibrated.
 303        """
 304        self.target: str = target
 305        """
 306        The name of the column in both `x_data` and `y_data` that
 307        will be used as the x and y variables in the calibration.
 308        """
 309        self.y_data = cont_strat_folds(
 310                y_data.loc[join_index, :],
 311                target,
 312                folds,
 313                strat_groups,
 314                seed
 315                )
 316        """
 317        The data that `x_data` will be calibrated against. A '*Fold*'
 318        column is added using the `const_strat_folds` function which splits
 319        the data into k stratified folds (where k is the value of
 320        `folds`). It splits the continuous measurements into n bins (where n
 321        is the value of `strat_groups`) and distributes each bin equally
 322        across all folds. This significantly reduces the chances of one fold
 323        containing a skewed distribution relative to the whole dataset.
 324        """
 325        self.models: dict[str,  # Technique name
 326                          dict[str,  # Scaling technique
 327                               dict[str,  # Variable combo
 328                                    dict[int,  # Fold
 329                                         Pipeline]]]] = dict()
 330        """
 331        The calibrated models. They are stored in a nested structure as
 332        follows:
 333        1. Primary Key, name of the technique (e.g Lasso Regression).
 334        2. Scaling technique (e.g Yeo-Johnson Transform).
 335        3. Combination of variables used or `target` if calibration is
 336        univariate (e.g "`target` + a + b).
 337        4. Fold, which fold was used excluded from the calibration. If data
 338        if 5-fold cross validated, a key of 4 indicates the data was trained on
 339        folds 0-3.
 340
 341        ```mermaid
 342            stateDiagram-v2
 343              models --> Technique
 344              state Technique {
 345                [*] --> Scaling
 346                [*]: The calibration technique used
 347                [*]: (e.g "Lasso Regression")
 348                state Scaling {
 349                  [*] --> Variables
 350                  [*]: The scaling technique used
 351                  [*]: (e.g "Yeo-Johnson Transform")
 352                  state Variables {
 353                    [*] : The combination of variables used
 354                    [*] : (e.g "x + a + b")
 355                    [*] --> Fold
 356                    state Fold {
 357                     [*] : Which fold was excluded from training data
 358                     [*] : (e.g 4 indicates folds 0-3 were used to train)
 359                    }
 360                  }
 361                }
 362              }
 363        ```
 364
 365        """

 367    def _sklearn_regression_meta(
 368            self,
 369            reg: Union[skl.base.RegressorMixin, Literal['t', 'gaussian']],
 370            name: str,
 371            scaler: Literal[
 372                'None',
 373                'Standard Scale',
 374                'MinMax Scale',
 375                'Yeo-Johnson Transform'
 376                'Box-Cox Transform',
 377                'Quantile Transform (Uniform)',
 378                'Quantile Transform (Gaussian)'
 379                ] = 'None',
 380            min_coeffs: int = 1,
 381            max_coeffs: int = (sys.maxsize * 2) + 1,
 382            **kwargs
 383            ):
 384        """
 385        Metaclass, formats data and uses sklearn classifier to
 386        fit x to y
 387
 388        Parameters
 389        ----------
 390        reg : sklearn.base.RegressorMixin or str
 391            Classifier to use, or distribution family to use for bayesian.
 392        name : str
 393            Name of classification technique to save pipeline to.
 394        min_coeffs : int, default=1
 395            Minimum number of coefficients for technique.
 396        max_coeffs : int, default=(sys.maxsize * 2) + 1
 397            Maximum number of coefficients for technique.
 398
 399        Raises
 400        ------
 401        NotImplementedError
 402            PyMC currently doesn't work, TODO
 403        """
 404        scalers: dict[str, Any] = {
 405                'None': None,
 406                'Standard Scale': pre.StandardScaler(),
 407                'MinMax Scale': pre.MinMaxScaler(),
 408                'Yeo-Johnson Transform': pre.PowerTransformer(
 409                    method='yeo-johnson'
 410                    ),
 411                'Box-Cox Transform': pre.PowerTransformer(method='box-cox'),
 412                'Quantile Transform (Uniform)': pre.QuantileTransformer(
 413                    output_distribution='uniform'
 414                    ),
 415                'Quantile Transform (Gaussian)': pre.QuantileTransformer(
 416                    output_distribution='normal'
 417                    )
 418                }
 419        x_secondary_cols = self.x_data.drop(self.target, axis=1).columns
 420        # All columns in x_data that aren't the target variable
 421        products = [[np.nan, col] for col in x_secondary_cols]
 422        secondary_vals = pd.MultiIndex.from_product(products)
 423        # Get all possible combinations of secondary variables in a pandas
 424        # MultiIndex
 425        if self.models.get(name) is None:
 426            self.models[name] = dict()
 427            # If the classification technique hasn't been used yet,
 428            # add its key to the models dictionary
 429        if self.models[name].get(scaler) is None:
 430            self.models[name][scaler] = dict()
 431            # If the scaling technique hasn't been used with the classification
 432            # technique yet, add its key to the nested dictionary
 433        for sec_vals in secondary_vals:
 434            # Loop over all combinations of secondary values
 435            vals = [self.target] + [v for v in sec_vals if v == v]
 436            vals_str = ' + '.join(vals)
 437            if len(vals) < min_coeffs or len(vals) > max_coeffs:
 438                # Skip if number of coeffs doesn't lie within acceptable range
 439                # for technique. For example, isotonic regression
 440                # only works with one variable
 441                continue
 442            self.models[name][scaler][vals_str] = dict()
 443            for fold in self.y_data.loc[:, 'Fold'].unique():
 444                y_data = self.y_data[
 445                        self.y_data.loc[:, 'Fold'] != fold
 446                        ]
 447                if reg in ['t', 'gaussian']:
 448                    # If using PyMC bayesian model,
 449                    # format data and build model using bambi
 450                    # then store result in pipeline
 451                    # Currently doesn't work as PyMC models
 452                    # can't be pickled, so don't function with deepcopy. Needs
 453                    # looking into
 454                    raise NotImplementedError(
 455                        "PyMC functions currently don't work with deepcopy"
 456                    )
 457#                    sc = scalers[scaler]
 458#                    if sc is not None:
 459#                        x_data = sc.fit_transform(
 460#                                self.x_data.loc[y_data.index, :]
 461#                                )
 462#                    else:
 463#                        x_data = self.x_data.loc[y_data.index, :]
 464#                    x_data['y'] = y_data.loc[:, self.target]
 465#                    model = bmb.Model(
 466#                            f"y ~ {vals_str}",
 467#                            x_data,
 468#                            family=reg
 469#                            )
 470#                    _ = model.fit(
 471#                        progressbar=False,
 472#                        **kwargs
 473#                        )
 474#                    pipeline = Pipeline([
 475#                        ("Scaler", scaler),
 476#                        ("Regression", model)
 477#                        ])
 478                else:
 479                    # If using scikit-learn API compatible classifier,
 480                    # Build pipeline and fit to
 481                    pipeline = Pipeline([
 482                        ("Selector", ColumnTransformer([
 483                                ("selector", "passthrough", vals)
 484                            ], remainder="drop")
 485                         ),
 486                        ("Scaler", scalers[scaler]),
 487                        ("Regression", reg)
 488                        ])
 489                    pipeline.fit(
 490                            self.x_data.loc[y_data.index, :],
 491                            y_data.loc[:, self.target]
 492                            )
 493                self.models[name][scaler][vals_str][fold] = dc(pipeline)

 495    def pymc_bayesian(

 42        df: pd.DataFrame,
 43        target_var: str,
 44        splits: int = 5,
 45        strat_groups: int = 5,
 46        seed: int = 62
 47        ) -> pd.DataFrame:
 48    """
 49    Creates stratified k-folds on continuous variable.
 50    Based off of work by tolgadincer [^kaggle].
 51
 52    [^kaggle]:
 53    www.kaggle.com/code/tolgadincer/continuous-target-stratification/notebook
 54
 55    Parameters
 56    ----------
 57    df : pd.DataFrame
 58        Target data to stratify on.
 59    target_var : str
 60        Target feature name.
 61    splits : int, default=5
 62        Number of folds to make.
 63    strat_groups : int, default=10
 64        Number of groups to split data in to for stratification.
 65    seed : int, default=62
 66        Random state to use.
 67    
 68    Returns
 69    -------
 70    pd.DataFrame
 71        `y_df` with added 'Fold' column, specifying which test data fold
 72        variable corresponds to.
 73
 74    Examples
 75    --------
 76    >>> df = pd.read_csv('data.csv')
 77    >>> df
 78    |    | x | a | b |
 79    |    |   |   |   |
 80    |  0 |2.3|1.8|7.2|
 81    |  1 |3.2|9.6|4.5|
 82    |....|...|...|...|
 83    |1000|2.3|4.5|2.2|
 84    >>> df_with_folds = const_strat_folds(
 85            df=df,
 86            target='a',
 87            splits=3,
 88            strat_groups=3.
 89            seed=78
 90        )
 91    >>> df_with_folds
 92    |    | x | a | b |Fold|
 93    |    |   |   |   |    |
 94    |  0 |2.3|1.8|7.2| 2  |
 95    |  1 |3.2|9.6|4.5| 1  |
 96    |....|...|...|...|....|
 97    |1000|2.3|4.5|2.2| 0  |
 98
 99    All folds should have a roughly equal distribution of values for 'a'
100
101    """
102    _df = df.copy()
103    _df['Fold'] = -1
104    skf = StratifiedKFold(
105            n_splits=splits,
106            random_state=seed,
107            shuffle=True
108            )
109    _df['Group'] = pd.cut(
110            _df.loc[:, target_var],
111            strat_groups,
112            labels=False
113            )
114    group_label = _df.loc[:, 'Group']
115
116    for fold_number, (_, v) in enumerate(skf.split(group_label, group_label)):
117        _df.loc[v, 'Fold'] = fold_number
118    return _df.drop('Group', axis=1)

 304        """
 305        self.target: str = target
 306        """
 307        The name of the column in both `x_data` and `y_data` that
 308        will be used as the x and y variables in the calibration.
 309        """
 310        self.y_data = cont_strat_folds(
 311                y_data.loc[join_index, :],
 312                target,
 313                folds,
 314                strat_groups,
 315                seed
 316                )
 317        """
 318        The data that `x_data` will be calibrated against. A '*Fold*'
 319        column is added using the `const_strat_folds` function which splits
 320        the data into k stratified folds (where k is the value of
 321        `folds`). It splits the continuous measurements into n bins (where n
 322        is the value of `strat_groups`) and distributes each bin equally
 323        across all folds. This significantly reduces the chances of one fold
 324        containing a skewed distribution relative to the whole dataset.
 325        """
 326        self.models: dict[str,  # Technique name
 327                          dict[str,  # Scaling technique
 328                               dict[str,  # Variable combo
 329                                    dict[int,  # Fold
 330                                         Pipeline]]]] = dict()
 331        """
 332        The calibrated models. They are stored in a nested structure as
 333        follows:
 334        1. Primary Key, name of the technique (e.g Lasso Regression).
 335        2. Scaling technique (e.g Yeo-Johnson Transform).
 336        3. Combination of variables used or `target` if calibration is
 337        univariate (e.g "`target` + a + b).
 338        4. Fold, which fold was used excluded from the calibration. If data
 339        if 5-fold cross validated, a key of 4 indicates the data was trained on
 340        folds 0-3.
 341
 342        ```mermaid
 343            stateDiagram-v2
 344              models --> Technique
 345              state Technique {
 346                [*] --> Scaling
 347                [*]: The calibration technique used
 348                [*]: (e.g "Lasso Regression")
 349                state Scaling {
 350                  [*] --> Variables
 351                  [*]: The scaling technique used
 352                  [*]: (e.g "Yeo-Johnson Transform")
 353                  state Variables {
 354                    [*] : The combination of variables used
 355                    [*] : (e.g "x + a + b")
 356                    [*] --> Fold
 357                    state Fold {
 358                     [*] : Which fold was excluded from training data
 359                     [*] : (e.g 4 indicates folds 0-3 were used to train)
 360                    }
 361                  }
 362                }
 363              }
 364        ```
 365
 366        """
 367
 368    def _sklearn_regression_meta(
 369            self,
 370            reg: Union[skl.base.RegressorMixin, Literal['t', 'gaussian']],
 371            name: str,
 372            scaler: Literal[
 373                'None',
 374                'Standard Scale',
 375                'MinMax Scale',
 376                'Yeo-Johnson Transform'
 377                'Box-Cox Transform',
 378                'Quantile Transform (Uniform)',
 379                'Quantile Transform (Gaussian)'
 380                ] = 'None',
 381            min_coeffs: int = 1,
 382            max_coeffs: int = (sys.maxsize * 2) + 1,
 383            **kwargs
 384            ):
 385        """
 386        Metaclass, formats data and uses sklearn classifier to
 387        fit x to y
 388
 389        Parameters
 390        ----------
 391        reg : sklearn.base.RegressorMixin or str
 392            Classifier to use, or distribution family to use for bayesian.
 393        name : str
 394            Name of classification technique to save pipeline to.
 395        min_coeffs : int, default=1
 396            Minimum number of coefficients for technique.
 397        max_coeffs : int, default=(sys.maxsize * 2) + 1
 398            Maximum number of coefficients for technique.
 399
 400        Raises
 401        ------
 402        NotImplementedError
 403            PyMC currently doesn't work, TODO
 404        """
 405        scalers: dict[str, Any] = {
 406                'None': None,
 407                'Standard Scale': pre.StandardScaler(),
 408                'MinMax Scale': pre.MinMaxScaler(),
 409                'Yeo-Johnson Transform': pre.PowerTransformer(
 410                    method='yeo-johnson'
 411                    ),
 412                'Box-Cox Transform': pre.PowerTransformer(method='box-cox'),
 413                'Quantile Transform (Uniform)': pre.QuantileTransformer(
 414                    output_distribution='uniform'
 415                    ),
 416                'Quantile Transform (Gaussian)': pre.QuantileTransformer(
 417                    output_distribution='normal'
 418                    )
 419                }
 420        x_secondary_cols = self.x_data.drop(self.target, axis=1).columns
 421        # All columns in x_data that aren't the target variable
 422        products = [[np.nan, col] for col in x_secondary_cols]
 423        secondary_vals = pd.MultiIndex.from_product(products)
 424        # Get all possible combinations of secondary variables in a pandas
 425        # MultiIndex
 426        if self.models.get(name) is None:
 427            self.models[name] = dict()
 428            # If the classification technique hasn't been used yet,
 429            # add its key to the models dictionary
 430        if self.models[name].get(scaler) is None:
 431            self.models[name][scaler] = dict()
 432            # If the scaling technique hasn't been used with the classification
 433            # technique yet, add its key to the nested dictionary
 434        for sec_vals in secondary_vals:
 435            # Loop over all combinations of secondary values
 436            vals = [self.target] + [v for v in sec_vals if v == v]
 437            vals_str = ' + '.join(vals)
 438            if len(vals) < min_coeffs or len(vals) > max_coeffs:
 439                # Skip if number of coeffs doesn't lie within acceptable range
 440                # for technique. For example, isotonic regression
 441                # only works with one variable
 442                continue
 443            self.models[name][scaler][vals_str] = dict()
 444            for fold in self.y_data.loc[:, 'Fold'].unique():
 445                y_data = self.y_data[
 446                        self.y_data.loc[:, 'Fold'] != fold
 447                        ]
 448                if reg in ['t', 'gaussian']:
 449                    # If using PyMC bayesian model,
 450                    # format data and build model using bambi
 451                    # then store result in pipeline
 452                    # Currently doesn't work as PyMC models
 453                    # can't be pickled, so don't function with deepcopy. Needs
 454                    # looking into
 455                    raise NotImplementedError(
 456                        "PyMC functions currently don't work with deepcopy"
 457                    )
 458#                    sc = scalers[scaler]
 459#                    if sc is not None:
 460#                        x_data = sc.fit_transform(
 461#                                self.x_data.loc[y_data.index, :]
 462#                                )
 463#                    else:
 464#                        x_data = self.x_data.loc[y_data.index, :]
 465#                    x_data['y'] = y_data.loc[:, self.target]
 466#                    model = bmb.Model(
 467#                            f"y ~ {vals_str}",
 468#                            x_data,
 469#                            family=reg
 470#                            )
 471#                    _ = model.fit(
 472#                        progressbar=False,
 473#                        **kwargs
 474#                        )
 475#                    pipeline = Pipeline([
 476#                        ("Scaler", scaler),
 477#                        ("Regression", model)
 478#                        ])
 479                else:
 480                    # If using scikit-learn API compatible classifier,
 481                    # Build pipeline and fit to
 482                    pipeline = Pipeline([
 483                        ("Selector", ColumnTransformer([
 484                                ("selector", "passthrough", vals)
 485                            ], remainder="drop")
 486                         ),
 487                        ("Scaler", scalers[scaler]),
 488                        ("Regression", reg)
 489                        ])
 490                    pipeline.fit(
 491                            self.x_data.loc[y_data.index, :],
 492                            y_data.loc[:, self.target]
 493                            )
 494                self.models[name][scaler][vals_str][fold] = dc(pipeline)
